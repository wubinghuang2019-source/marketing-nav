  <script>

    let sitesData = null;

    let activeCategory = 'all';

    async function loadData() {

      const response = await fetch('sites.json');

      sitesData = await response.json();

      renderCategoryNav();

      renderSites();

      document.getElementById('updateDate').textContent = sitesData.meta.updatedAt;

    }

    function

renderCategoryNav() {

      document.getElementById('categoryNav').innerHTML = 

        <button class="category-btn active" onclick="filterCategory('all')">全部</button>

        ${sitesData.categories.map(cat => 

          <button class="category-btn" onclick="filterCategory('${cat.id}')">${cat.icon} ${cat.name}</button>

        ).join('')}

      ;

    }

    function renderSites(filterText = '') {

      const container = document.getElementById('sitesContainer');

      if (activeCategory === 'all') {

        container.innerHTML = sitesData.categories.map(cat => renderCategorySection(cat, filterText)).filter(h=>h).join('');

      } else {

        const cat = sitesData.categories.find(c => c.id === activeCategory);

        if (cat) container.innerHTML = renderCategorySection(cat, filterText);

}

    }

    function renderCategorySection(cat, filterText = '') {

      let sites = [...cat.sites];

      if (cat.sortBy === 'clicks') sites.sort((a, b) => (b.clicks || 0) - (a.clicks || 0));

      const filtered = sites.filter(site => {

        if (!filterText) return true;

        return (site.name + site.desc + site.tags.join(' ')).toLowerCase().includes(filterText.toLowerCase());

      });

      if (filtered.length === 0) return '';

      return 

        <div class="category-section" style="margin:40px 0;">

          <h2 style="font-size:24px;color:#333;margin-bottom:20px;">${cat.icon} ${cat.name}</h2>

          <div class="sites-grid">

            ${filtered.map(site => 

              <div class="site-card" onclick="trackClick('${cat.id}','${site.name}')">

                <a href="${site.url}"

target="_blank" style="text-decoration:none;color:inherit;display:block;">

                  <div class="site-header">

                    <div class="site-icon">🔗</div>

                    <div>

                      <div class="site-name">${site.name}</div>

                      <div style="font-size:12px;color:#999;">${new URL(site.url!=='#'?site.url:'https://example.com').hostname}</div>

                    </div>

                  </div>

                  <div class="site-desc">${site.desc}</div>

                  <div>${site.tags.map(tag => <span class="tag">${tag}</span>).join('')}</div>

                </a >

              </div>

            ).join('')}

          </div>

        </div>

      ;

    }

    function trackClick(categoryId, siteName) {

      let clicks = JSON.parse(localStorage.getItem('navClicks') || '{}');

const key = ${categoryId}_${siteName};

      clicks[key] = (clicks[key] || 0) + 1;

      localStorage.setItem('navClicks', JSON.stringify(clicks));

      if (categoryId === 'hot') renderSites();

    }

    function filterCategory(id) {

      activeCategory = id;

      document.querySelectorAll('.category-btn').forEach(btn => btn.classList.toggle('active', btn.textContent.includes(id)));

      renderSites();

    }

    function handleSearch() {

      const text = document.getElementById('searchInput').value;

      if (text) {

        document.getElementById('sitesContainer').innerHTML = sitesData.categories.map(cat => renderCategorySection(cat, text)).filter(h=>h).join('');

      } else renderSites();

    }

    loadData();
